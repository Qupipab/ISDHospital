USE ISDHospital

SELECT
    d.firstName + ' ' + d.lastName + ' ' + d.patron AS FIO, dp.title AS [Department Name], begBreak, DATEDIFF(minute, begBreak, endBreak) AS [Break], endBreak
FROM  WORKINGHOURS AS wh 
INNER JOIN DOCTOR AS d 
    ON wh.doctorID = d.doctorID
INNER JOIN department AS dp 
    ON d.departmentID = dp.departNum
WHERE
	DATEDIFF(minute, begBreak, endBreak) > 30
ORDER BY
	[Break] DESC

SELECT
(SELECT COUNT(beginning)
	FROM appointment
	WHERE DATEPART(HOUR, beginning) BETWEEN 7 AND 10) AS [7 - 10],

(SELECT COUNT(beginning)
	FROM appointment
	WHERE DATEPART(HOUR, beginning) BETWEEN 11 AND 13) AS [10 - 13],

(SELECT COUNT(beginning)
	FROM appointment
	WHERE DATEPART(HOUR, beginning) BETWEEN 14 AND 16) AS [13 - 16],

(SELECT COUNT(beginning)
	FROM appointment
	WHERE DATEPART(HOUR, beginning) BETWEEN 17 AND 19)  AS [16 - 19]
